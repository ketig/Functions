🔧 ფუნქციების სახეები

JavaScript-ში ფუნქციების 3 მთავარი სახე:

  • Function Declaration - ფუნქციის გამოცხადება
  • Function Expression - ფუნქციის გამოსახულება
  • Arrow Function - ისრისებრი ფუნქცია (ES6)

1️⃣ Function Declaration


// Function Declaration
function greetUser(name) {
    return `Hello, ${name}!`;
}

// ფუნქციის გამოძახება გამოცხადებამდეც შეიძლება (Hoisting)
console.log(sayHi()); // "Hi there!"

function sayHi() {
    return "Hi there!";
}

2️⃣ Function Expression

// Function Expression
const [sum, difference, product, quotient] = calculate(10, 5);
console.log(`Sum: ${sum}, Difference: ${difference}`);
// Sum: 15, Difference: 5

// Nested arrays
const matrix = [[1, 2], [3, 4]];
const [[a, b], [c, d]] = matrix;
console.log(a, b, c, d); // 1 2 3 4

const add = (a, b) => a + b;

const getRandomNumber = () => Math.random();

                greetUser = function(name) {
return `Hello, ${name}!`;
};

🎯 Scope საფუძვლები

რა არის Scope?

Scope განსაზღვრავს სად არის ხელმისაწვდომი ცვლადები და ფუნქციები კოდში. JavaScript-ში არის რამდენიმე სახის scope:

  • Global Scope - გლობალური
  • Function Scope - ფუნქციის
  • Block Scope - ბლოკის (ES6+)

🌍 Global Scope

// Global Scope - ყველგან ხელმისაწვდომი
let globalVar = "I'm global!";
var oldGlobal = "I'm also global!";

function showGlobal() {
    console.log(globalVar); // ხელმისაწვდომია
    console.log(oldGlobal); // ხელმისაწვდომია
}

showGlobal(); // "I'm global!" "I'm also global!"

🔧 Function Scope

function myFunction() {
    let functionVar = "I'm in function scope";
    var oldFunctionVar = "Me too!";
    
    console.log(functionVar); // მუშაობს
    console.log(oldFunctionVar); // მუშაობს
}

myFunction();

// console.log(functionVar); // Error! არ არის ხელმისაწვდომი
// console.log(oldFunctionVar); // Error! არ არის ხელმისაწვდომი

📦 Block Scope (ES6+)

// Block Scope - let და const გამოყენებისას
if (true) {
    let blockVar = "I'm in block scope";
    const blockConst = "Me too!";
    var oldVar = "I'm not!";
    
    console.log(blockVar); // მუშაობს
    console.log(blockConst); // მუშაობს
}

// console.log(blockVar); // Error!
// console.log(blockConst); // Error!
console.log(oldVar); // მუშაობს! var არ იცნობს block scope-ს

// for loop-ში block scope
for (let i = 0; i < 3; i++) {
    console.log(i); // 0, 1, 2
}
// console.log(i); // Error!
💡 რჩევა: ყოველთვის გამოიყენეთ let და const var-ის ნაცვლად!

🔗 Lexical Scope & Scope Chain

Lexical Scope

Lexical Scope ნიშნავს, რომ ფუნქციის scope განისაზღვრება იქ, სადაც ის არის განსაზღვრული კოდში, არა იქ სადაც იძახება.

let globalVar = "Global";

function outerFunction() {
    let outerVar = "Outer";
    
    function innerFunction() {
        let innerVar = "Inner";
        
        // ყველა ცვლადი ხელმისაწვდომია
        console.log(innerVar);  // "Inner"
        console.log(outerVar);  // "Outer"
        console.log(globalVar); // "Global"
    }
    
    innerFunction();
    // console.log(innerVar); // Error! არ არის ხელმისაწვდომი
}

outerFunction();

🔗 Scope Chain

JavaScript ეძებს ცვლადებს Scope Chain-ის მიხედვით: ჯერ ლოკალურ scope-ში, შემდეგ parent scope-ში და ასე შემდეგ global scope-მდე.

let name = "Global Ana";

function level1() {
    let name = "Level1 Nino";
    
    function level2() {
        let name = "Level2 Mari";
        
        function level3() {
            // name-ს არ აქვს ლოკალური განსაზღვრება
            console.log(name); // "Level2 Mari" - ახლო scope-იდან
        }
        
        level3();
    }
    
    level2();
}

level1();

// მაგალითი scope chain-ზე
function findVariable() {
    // let searchVar = "local"; // ამის კომენტარი გაუკეთეთ
    
    function inner() {
        console.log(searchVar); // ეძებს parent scope-ებში
    }
    
    inner();
}

let searchVar = "global";
findVariable(); // "global"
🎯 მნიშვნელოვანი: JavaScript ყოველთვის იყენებს ახლო scope-ს. თუ ცვლადი არსებობს რამდენიმე scope-ში, გამოიყენება ყველაზე ახლო.

📦 Closures

რა არის Closure?

Closure ეს არის ფუნქცია, რომელსაც აქვს წვდომა თავის parent scope-ის ცვლადებზე, მაშინაც კი როცა parent ფუნქცია უკვე დასრულებულია.

🔮 ძირითადი მაგალითი

function createCounter() {
    let count = 0; // private ცვლადი
    
    return function() {
        count++; // შიდა ფუნქცია ახსოვს parent-ის ცვლადს
        return count;
    };
}

const counter1 = createCounter();
const counter2 = createCounter();

console.log(counter1()); // 1
console.log(counter1()); // 2
console.log(counter2()); // 1 (განცალკევებული counter)
console.log(counter1()); // 3

🛡️ Private ცვლადები

function createBankAccount(initialBalance) {
    let balance = initialBalance;
    
    return {
        deposit: function(amount) {
            balance += amount;
            return balance;
        },
        
        withdraw: function(amount) {
            if (amount <= balance) {
                balance -= amount;
                return balance;
            }
            return "არასაკმარისი თანხა";
        },
        
        getBalance: function() {
            return balance;
        }
    };
}

const myAccount = createBankAccount(1000);

console.log(myAccount.getBalance()); // 1000
console.log(myAccount.deposit(500));  // 1500
console.log(myAccount.withdraw(200)); // 1300

// balance-ზე პირდაპირ წვდომა შეუძლებელია
// console.log(myAccount.balance); // undefined

✨ Rest & Spread Operators

Rest (...) Operator

Rest operator აგროვებს მრავალ ელემენტს ერთ მასივში. გამოიყენება ფუნქციის პარამეტრებში და destructuring-ში.

Spread (...) Operator

Spread operator "აშლის" მასივს ან ობიექტს ცალკეულ ელემენტებად.

📥 Rest Parameters

// Rest parameters - ყველა argument-ი მასივში
function sum(...numbers) {
    console.log(numbers); // მასივია
    return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3, 4, 5)); // 15
console.log(sum(10, 20)); // 30

// Rest-ი უნდა იყოს ბოლოში
function greetUsers(greeting, ...names) {
    return names.map(name => `${greeting}, ${name}!`);
}

console.log(greetUsers("Hello", "Ana", "Nino", "Mari"));
// ["Hello, Ana!", "Hello, Nino!", "Hello, Mari!"]

📤 Spread Syntax

// Spread მასივებთან
const numbers1 = [1, 2, 3];
const numbers2 = [4, 5, 6];

// მასივების გაერთიანება
const combined = [...numbers1, ...numbers2];
console.log(combined); // [1, 2, 3, 4, 5, 6]

// მასივის კოპირება
const copied = [...numbers1];
console.log(copied); // [1, 2, 3]

// Math.max()-ით გამოყენება
console.log(Math.max(...numbers1)); // 3

// Spread ობიექტებთან (ES2018)
const person = { name: "Ana", age: 25 };
const address = { city: "Tbilisi", country: "Georgia" };

const fullInfo = { ...person, ...address, age: 26 };
console.log(fullInfo);
// { name: "Ana", age: 26, city: "Tbilisi", country: "Georgia" }

🔄 Rest vs Spread

// REST - აგროვებს
function processItems(...items) { // REST
    console.log("Items:", items);
    return items.length;
}

const myArray = [1, 2, 3, 4, 5];

// SPREAD - აშლის
const count = processItems(...myArray); // SPREAD
console.log("Count:", count); // 5

// Array destructuring-ში REST
const [first, second, ...rest] = myArray;
console.log("First:", first);   // 1
console.log("Second:", second); // 2
console.log("Rest:", rest);     // [3, 4, 5]

📋 Array Destructuring

Array Destructuring

Array Destructuring გვაძლევს საშუალებას მასივის ელემენტები ცალკეულ ცვლადებში გავანაწილოთ მოკლე და მარტივი სინტაქსით.

🎯 ძირითადი გამოყენება

// ძველი გზა
const fruits = ["apple", "banana", "orange"];
const firstFruit = fruits[0];
const secondFruit = fruits[1];

// ახალი გზა - Destructuring
const [first, second, third] = fruits;

console.log(second); // "banana"
console.log(third);  // "orange"

// ელემენტების გამოტოვება

console.log(c); // "orange"

📁 ES6 Modules

ES6 Modules

Modules საშუალებას გვაძლევს კოდი დავყოთ ცალკეულ ფაილებში და მოდულებში. ეს ხდის კოდს უფრო ორგანიზებულს, ხელახლა გამოსაყენებელს და მარტივად სატესტოს.

📤 Export - ექსპორტი

// math.js ფაილი
// Named exports
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export function multiply(a, b) {
    return a * b;
}

// ან ერთად export
const subtract = (a, b) => a - b;
const divide = (a, b) => a / b;

export { subtract, divide };

// Default export
export default function calculator(operation, a, b) {
    switch(operation) {
        case 'add': return add(a, b);
        case 'subtract': return subtract(a, b);
        case 'multiply': return multiply(a, b);
        case 'divide': return divide(a, b);
        default: return 0;
    }
}

📥 Import - იმპორტი

// main.js ფაილი
// Named imports
import { PI, add, multiply } from './math.js';

console.log(PI); // 3.14159
console.log(add(5, 3)); // 8

// Default import
import calculator from './math.js';
console.log(calculator('multiply', 4, 5)); // 20

// Import all as namespace
import * as MathUtils from './math.js';
console.log(MathUtils.PI); // 3.14159
console.log(MathUtils.add(2, 3)); // 5

// Import with alias
import { subtract as minus, divide as div } from './math.js';
console.log(minus(10, 3)); // 7

// Mixed imports
import calculator, { PI, add } from './math.js';

};

export class Logger {
    static log(message) {
        console.log(`[${new Date().toISOString()}] ${message}`);
    }
    
    static error(message) {
        console.error(`[ERROR] ${message}`);
    }
}
